5   /**
6    * The Trie class, which is a data structure for holding valid words.
7    */
8   public class Trie {
9     private TrieNode root;
10  
11    /**
12     * Constructor.
13     */
14    public Trie() {
15      root = new TrieNode();
16    }
17  
18    /**
19     * Adds a valid word to the Trie.
20     *
21     * @param newWord The new word to add.
22     */
23    public void addValidWord(String newWord) {
24      TrieNode currNode = root;
25      newWord = newWord.toLowerCase();
26  
27      for (String letter : newWord.split("")) {
28        if (!currNode.hasChild(letter)) {
29          currNode.addChild(letter, new TrieNode());
30        }
31  
32        // update the child with the next one, continuing until the word is done.
33        currNode = currNode.getChild(letter);
34      }
35  
36      currNode.setEndOfWord(true);
37    }
38  
39    /**
40     * Returns whether the word is valid.
41     *
42     * @param word The word to check.
43     * @return Whether the word has been added to the trie.
44     */
45    public boolean isValidWord(String word) {
46      TrieNode currNode = root;
47      word = word.toLowerCase();
48  
49      for (String letter : word.split("")) {
50        if (currNode.hasChild(letter)) {
51          currNode = currNode.getChild(letter);
52        } else {
53          // if the current node doesn't have the next letter as child, then we
54          // automatically know that it isn't word since it hasn't been logged
55          // before.
56          return false;
57        }
58      }
59  
60      return currNode.isEndOfWord();
61    }
62  
63    /**
64     * Returns a list of strings. Index 0 of the array means it's an exact
65     * match, and index 1 means there was one LED, and so forth. It will fill
66     * each equivalence class.
67     *
68     * @param led        The maximum LED.
69     * @param word       The word to autocorrect.
70     * @param whitespace Whether whitespace is on.
71     * @param prefix     Whether prefix-completion is on.
72     * @return List of autocorrected suggestions.
73     */
74    public ArrayList<String> autoCorrect(int led, String word,
75                                         boolean whitespace, boolean prefix) {
76      ArrayList<String> suggestions = new ArrayList();
77  
78      search(suggestions, root, "", word, 0, led);
79  
80      // check for whitespace: the max LED must be at least 1 since inserting a
81      // space counts as an edit.
82      if (whitespace) {
83        whitespaceSplit(suggestions, root, "", word, false);
84      }
85  
86      // also append everything that also starts with this
87      if (prefix) {
88        TrieNode start = getNodeUntil(word);
89        if (start != null) {
90          autocomplete(suggestions, start, word);
91        }
92      }
93  
94      return suggestions;
95    }
96  
97    /**
98     * Autocompletes a word. Recursive.
99     *
100    * @param suggestions List of already found suggestions.
101    * @param node        Node we're currently on.
102    * @param wordSoFar   Word constructed so far.
103    */
104   private void autocomplete(ArrayList<String> suggestions, TrieNode node,
105                             String wordSoFar) {
106     if (node.isEndOfWord() && !suggestions.contains(wordSoFar)) {
107       suggestions.add(wordSoFar);
108     }
109 
110     for (String child : node.getChildChars()) {
111       TrieNode childNode = node.getChild(child);
112       autocomplete(suggestions, childNode, wordSoFar + child);
113     }
114   }

116   /**
117    * Recursive method for traversing the trie looking for suggestions.
118    *
119    * @param foundWords The array of found words, where each array index
120    *                   indicates the LED for that suggestion.
121    * @param node       The current node to search.
122    * @param remainder  The rest of the word to find, starting with the next
123    *                   letter.
124    * @param currLED    Number of edits made so far in the string
125    * @param maxLED     Maximum number of edits allowed.
126    */
127   private void search(ArrayList<String> foundWords, TrieNode node,
128                       String currWord, String remainder, int currLED,
129                       int maxLED) {
130     // base case
131     if (currLED > maxLED) {
132       return;
133     }
134 
135     // if there's nothing left in the remainder, then we can check for adds
136     // and also see if we can add this word to our list.
137     if (remainder.length() == 0) {
138       // add to list if it's a word!
139       if (node.isEndOfWord() && currWord.length() > 0
140           && !foundWords.contains(currWord)) {
141         foundWords.add(currWord);
142       }
143 
144       // also check for special case addition
145       for (String key : node.getChildChars()) {
146         String updatedWord = currWord + key;
147         search(foundWords, node.getChild(key), updatedWord, remainder,
148             currLED + 1, maxLED);
149       }
150     } else {
151       TrieNode correctNextChild = null;
152       String nextChar = remainder.substring(0, 1);
153 
154       // if correct next letter exists, there's no change to the LED for
155       // changing the letter
156       if (node.hasChild(nextChar)) {
157         correctNextChild = node.getChild(nextChar);
158         search(foundWords, correctNextChild, currWord + nextChar,
159             remainder.substring(1), currLED, maxLED); // changing a letter
160         // (no change to LED)
161         search(foundWords, correctNextChild, currWord + nextChar, remainder,
162             currLED + 1, maxLED); // adding a letter
163         search(foundWords, node, currWord, remainder.substring(1),
164             currLED + 1, maxLED); // subtracting a letter
165       }
166 
167       for (String childChar : node.getChildChars()) {
168         TrieNode childNode = node.getChild(childChar);
169 
170         // make sure this isn't the one we've already searched
171         if (correctNextChild == null || !childNode.equals(correctNextChild)) {
172           search(foundWords, childNode, currWord + childChar,
173               remainder.substring(1), currLED + 1, maxLED); // changing a letter
174           search(foundWords, childNode, currWord + childChar, remainder,
175               currLED + 1, maxLED); // adding a letter
176           search(foundWords, node, currWord, remainder.substring(1),
177               currLED + 1, maxLED); // subtracting a letter
178         }
179       }
180     }
181   }

183   /**
184    * Looks for matches with whitespace splitting enabled. Recursive.
185    *
186    * @param foundWords   List of found words.
187    * @param node         Starting node.
188    * @param currWord     The current word we've constructed.
189    * @param remainder    The remaining of the word we have.
190    * @param alreadySplit Whether we've already made a split.
191    */
192   private void whitespaceSplit(ArrayList<String> foundWords, TrieNode node,
193                                String currWord, String remainder,
194                                boolean alreadySplit) {
195     // first find out if this is actually a valid word we can split on
196 
197     if (node.isEndOfWord()) {
198       if (remainder.length() == 0 && !foundWords.contains(currWord)) {
199         // if we've split already and there's nothing left in the remainder,
200         // this is a valid suggestion.
201         foundWords.add(currWord);
202       } else if (!alreadySplit) {
203         whitespaceSplit(foundWords, root, currWord + " ", remainder, true);
204       }
205     }
206 
207     if (remainder.length() > 0) {
208       // if this is not a word, we keep going along the trie, looking for
209       // exact matches only
210       String nextChar = remainder.substring(0, 1);
211 
212       if (node.hasChild(nextChar)) {
213         TrieNode nextChild = node.getChild(nextChar);
214         whitespaceSplit(foundWords, nextChild, currWord + nextChar,
215             remainder.substring(1), alreadySplit);
216       }
217     }
218   }

220   /**
221    * Returns the node up until a certain string has been reached.
222    *
223    * @param word The word/prefix to find.
224    * @return Returns null if a child node is not found. (aka doesn't exist.)
225    */
226   private TrieNode getNodeUntil(String word) {
227     TrieNode currNode = root;
228     word = word.toLowerCase();
229 
230     for (String letter : word.split("")) {
231       if (currNode.hasChild(letter)) {
232         currNode = currNode.getChild(letter);
233       } else {
234         // if the current node doesn't have the next letter as child, then we
235         // automatically know that it isn't word since it hasn't been logged
236         // before.
237         return null;
238       }
239     }
240 
241     return currNode;
242   }
