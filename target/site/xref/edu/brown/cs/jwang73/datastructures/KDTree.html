6   /**
7    * The KDTree data structure class. Each node must be a KDTreeComparable
8    * with the same dimensions listed in the same order. Call build to build
9    * the KD tree before using it.
10   * @param <T> The type of element to hold.
11   */
12  public class KDTree<T extends KDTreeComparable> {
13    private ArrayList<T> nodes;
14    private int expectedDimensions; // a precaution that should be constant
15  
16    private KDTreeNode<T> rootNode;
17  
18    /**
19     * Constructor.
20     */
21    public KDTree() {
22      nodes = new ArrayList<>();
23    }
24  
25    /**
26     * Adds a node to the KDTree.
27     *
28     * @param node node to add
29     */
30    public void add(T node) {
31      if (nodes.isEmpty()) {
32        expectedDimensions = node.getDimensions().length;
33      } else {
34        assert node.getDimensions().length == expectedDimensions;
35      }
36      nodes.add(node);
37    }
38  
39    /**
40     * Returns the number of elements in the tree.
41     *
42     * @return size of the tree
43     */
44    public int size() {
45      return nodes.size();
46    }
47  
48    /**
49     * Returns the list of stored elements.
50     * @return list of elements.
51     */
52    public ArrayList<T> getElements() {
53      return nodes; // could probably return a readonly version for encapsulation?
54    }
55  
56    /**
57     * Builds the KDTree.
58     */
59    public void build() {
60      if (nodes.isEmpty()) {
61        return;
62      }
63      rootNode = buildTree(nodes, 0);
64    }
65  
66    /**
67     * Returns the root node of the tree.
68     *
69     * @return the root node
70     */
71    public KDTreeNode<T> getRoot() {
72      return rootNode;
73    }
74  
75    /**
76     * Finds the n nearest neighbors around a source. The source does not have to
77     * already be in the KDTree, but it is excluded from the list of results.
78     *
79     * @param source       the node to find neighbors around
80     * @param numNeighbors the number of neighbors to find
81     * @return an ArrayList of the n nearest neighbors, sorted from close to far.
82     */
83    public ArrayList<T> findNearestNeighbors(T source, int numNeighbors) {
84      assert rootNode != null : "ERROR: Please build the tree first!";
85      assert numNeighbors >= 0 : "ERROR: Neighbors must be non-negative!";
86      if (numNeighbors == 0) {
87        return new ArrayList<>(); // return empty
88      }
89  
90      ArrayList<T> foundNeighbors = new ArrayList<>();
91      findNeighbors(rootNode, source, foundNeighbors, numNeighbors, 0);
92      sort(foundNeighbors, source);
93  
94      return foundNeighbors;
95    }
96  
97    /**
98     * Finds the nodes within a nonnegative radius of a source. The source does
99     * not have to be already in the KDTree.
100    *
101    * @param source the source to search around
102    * @param radius the radius to search in
103    * @return an ArrayList of the stars in the radius, sorted from close to far.
104    */
105   public ArrayList<T> findNodesInRadius(T source, double radius) {
106     assert radius >= 0 : "ERROR: Radius must be nonnegative!";
107     assert rootNode != null : "ERROR: Please build the tree first!";
108 
109     ArrayList<T> foundInRadius = new ArrayList<>();
110     findInRadius(rootNode, source, foundInRadius, radius, 0);
111     sort(foundInRadius, source);
112 
113     return foundInRadius;
114   }

116   /**
117    * Returns a KDTreeNode based on the list of nodes, and dimensionIndex is
118    * the dimension to search on. The public build method calls this on the
119    * whole set. This method will recurse on itself to build the tree.
120    *
121    * @param nodesToBuild the nodes left to build with
122    * @param axis  the relevant axis this time
123    * @return the built KDTreeNode
124    */
125   private KDTreeNode<T> buildTree(ArrayList<T> nodesToBuild, int axis) {
126     // base cases
127     if (nodesToBuild.isEmpty()) {
128       return null;
129     } else if (nodesToBuild.size() == 1) {
130       KDTreeNode<T> node = new KDTreeNode<>();
131       node.setElement(nodesToBuild.get(0));
132       return node;
133     }
134 
135     axis = axis % expectedDimensions;
136     double median = findMedian(nodesToBuild, axis);
137 
138     ArrayList<T> leftNodes = new ArrayList<>();
139     ArrayList<T> rightNodes = new ArrayList<>();
140     // we want to find the medianNode somewhere so it's left out of the next set
141     T medianNode = null;
142 
143     for (T node : nodesToBuild) {
144       if (node.getDimensions()[axis] > median) {
145         rightNodes.add(node);
146       } else if (node.getDimensions()[axis] < median) {
147         leftNodes.add(node);
148       } else {
149         if (medianNode == null) {
150           medianNode = node;
151           continue;
152         }
153         // if there are multiple nodes with the same median, then add it to the
154         // smaller bucket
155         ArrayList<T> smallerCollection = leftNodes.size() < rightNodes.size()
156             ? leftNodes : rightNodes;
157         smallerCollection.add(node);
158       }
159     }
160 
161     KDTreeNode<T> root = new KDTreeNode<>();
162     root.setElement(medianNode);
163 
164     root.setLeft(buildTree(leftNodes, axis + 1));
165     root.setRight(buildTree(rightNodes, axis + 1));
166 
167     return root;
168   }

170   /**
171    * Returns the median out of a collection of nodes, in the axis index.
172    *
173    * @param n          nodes to find the median of
174    * @param dimensionIndex the axis to search around
175    * @return the median
176    */
177   private double findMedian(ArrayList<T> n, int dimensionIndex) {
178     // build a list
179     ArrayList<Double> values = new ArrayList<>();
180     for (T node : n) {
181       values.add(node.getDimensions()[dimensionIndex]);
182     }
183     Collections.sort(values);
184 
185     return values.get(values.size() / 2);
186   }

188   /**
189    * Helper method to the neighbors method. Takes in a KDTreeNode and searches
190    * it and all its children. The ArrayList passed in will hold the neighbors
191    * found.
192    *
193    * @param currNode       the current node we're searching
194    * @param source         the target node
195    * @param foundNeighbors list of neighbors already found
196    * @param depth          level of recursion, used to find the axis
197    */
198   private void findNeighbors(KDTreeNode<T> currNode, T source,
199                              ArrayList<T> foundNeighbors, int numToFind,
200                              int depth) {
201     // base case.
202     if (currNode == null) {
203       return;
204     }
205 
206     T furthestFoundNeighbor = null;
207 
208     if (!currNode.getElement().equals(source)) {
209 
210       // if we don't currently have enough, add this node.
211       if (foundNeighbors.size() < numToFind) {
212         foundNeighbors.add(currNode.getElement());
213 
214         // otherwise, check against the other neighbors, but only if the current
215         // node isn't the original source.
216       } else {
217         double currDistance = getDistance(currNode.getElement().getDimensions(),
218             source.getDimensions());
219         furthestFoundNeighbor = foundNeighbors.get(0);
220 
221         // first find the furthest neighbor
222         for (T n : foundNeighbors) {
223           double currNeighborDistance = getDistance(n.getDimensions(),
224               source.getDimensions());
225           if (currNeighborDistance
226               > getDistance(furthestFoundNeighbor.getDimensions(),
227                   source.getDimensions())) {
228             furthestFoundNeighbor = n;
229           }
230         }
231 
232         // now see if it's further away than where we currently are
233         if (getDistance(furthestFoundNeighbor.getDimensions(),
234             source.getDimensions()) > currDistance) {
235           foundNeighbors.remove(furthestFoundNeighbor);
236           foundNeighbors.add(currNode.getElement());
237         }
238       }
239     }
240 
241     int axis = depth % expectedDimensions;
242 
243     // now we see which side we should recur on
244     boolean recurRight = false;
245     boolean recurLeft = false;
246 
247     if (currNode.getElement().getDimensions()[axis]
248         < source.getDimensions()[axis]) {
249       recurRight = true;
250     } else {
251       recurLeft = true;
252     }
253 
254     // check if we also need to recur on the other child
255     if (furthestFoundNeighbor != null) {
256       double furthestDistance = getDistance(
257           furthestFoundNeighbor.getDimensions(), source.getDimensions());
258       if (furthestDistance > Math.abs(source.getDimensions()[axis]
259           - currNode.getElement().getDimensions()[axis])) {
260         // recur on other child
261         recurLeft = true;
262         recurRight = true;
263       }
264     } else if (foundNeighbors.size() < numToFind) {
265       recurLeft = true;
266       recurRight = true;
267     }
268 
269     if (recurLeft) {
270       findNeighbors(currNode.getLeft(), source, foundNeighbors, numToFind,
271           depth + 1);
272     }
273 
274     if (recurRight) {
275       findNeighbors(currNode.getRight(), source, foundNeighbors, numToFind,
276           depth + 1);
277     }
278   }

280   /**
281    * Helper method to the radius method. Takes in a KDTreeNode and searches it
282    * and all its children. The ArrayList passed in will hold the found nodes.
283    *
284    * @param currNode the current node being searched
285    * @param source   the target
286    * @param found    the list of found nodes
287    * @param radius   the radius to search in
288    * @param depth    level of recursion
289    */
290   private void findInRadius(KDTreeNode<T> currNode, T source,
291                             ArrayList<T> found, double radius, int depth) {
292     // base case.
293     if (currNode == null) {
294       return;
295     }
296 
297     // if the current node is indeed within the radius, add it to the list.
298     if (!currNode.getElement().equals(source)
299         && getDistance(currNode.getElement().getDimensions(),
300             source.getDimensions()) <= radius) {
301       found.add(currNode.getElement());
302     }
303 
304     int axis = depth % expectedDimensions;
305     double distanceToSourceOnAxis = Math.abs(
306         currNode.getElement().getDimensions()[axis]
307             - source.getDimensions()[axis]);
308     boolean recurLeft = true;
309     boolean recurRight = true;
310 
311     // if it's further than the radius on an axis, then we can eliminate
312     // the other side entirely.
313     if (distanceToSourceOnAxis >= radius) {
314       if (currNode.getElement().getDimensions()[axis]
315           < source.getDimensions()[axis]) {
316         recurLeft = false;
317       } else {
318         recurRight = false;
319       }
320     }
321 
322     if (recurLeft) {
323       findInRadius(currNode.getLeft(), source, found, radius, depth + 1);
324     }
325 
326     if (recurRight) {
327       findInRadius(currNode.getRight(), source, found, radius, depth + 1);
328     }
329   }

331   /**
332    * Returns the distance between two points. Assumes that the two are the same
333    * length, given the precautions we took when adding nodes to the KDTree.
334    *
335    * @param a point a's coordinates in an array
336    * @param b point b's coordinates in an array
337    * @return the distance
338    */
339   private double getDistance(double[] a, double[] b) {
340     double sum = 0;
341     for (int i = 0; i < a.length; i++) {
342       double diff = Math.abs(a[i] - b[i]);
343       sum += diff * diff;
344     }
345     return Math.sqrt(sum);
346   }

348   /**
349    * Sorts the list given. For simplicity's sake, we use a slower but easier
350    * sort, assuming that the number of elements is usually small enough to
351    * be constant.
352    *
353    * @param list   list to sort
354    * @param source target to sort around based on distance
355    */
356   private void sort(ArrayList<T> list, T source) {
357     for (int i = 1; i < list.size(); i++) {
358       // visit each element before the current index
359       for (int j = 0; j < i; j++) {
360         double currDistance = getDistance(source.getDimensions(),
361             list.get(i).getDimensions());
362         double distanceToCompare = getDistance(source.getDimensions(),
363             list.get(j).getDimensions());
364         if (currDistance < distanceToCompare) {
365           T holder = list.get(i);
366           list.remove(i);
367           list.add(j, holder);
368         }
369       }
370     }
371   }
