1   package edu.brown.cs.jwang73.csvparser;
2   
3   import org.junit.Test;
4   
5   import static org.junit.Assert.assertNotNull;
6   import static org.junit.Assert.assertTrue;
7   
8   /**
9    * JUnit testing for the CSVParser.
10   */
11  public class CSVParserTest {
12  
13    /**
14     * Tests that it doesn't return null.
15     */
16    @Test
17    public void notNullTest() {
18      assertNotNull(CSVParser.parse("hello world!"));
19    }
20  
21    /**
22     * Tests a simple line.
23     */
24    @Test
25    public void oneLineTest() {
26      String s = "hello,world,!";
27      String[][] parsed = CSVParser.parse(s);
28      assertTrue(parsed[0][0].equals("hello"));
29      assertTrue(parsed[0][2].equals("!"));
30      assertTrue(parsed.length == 1);
31      assertTrue(parsed[0].length == 3);
32    }
33  
34    /**
35     * Tests multiple lines.
36     */
37    @Test
38    public void multiLineTest() {
39      String s = "the,quick,brown,fox\n"
40          + "jumps,over,the,lazy\n"
41          + "dog,blah,and so forth,yes yes no";
42      String[][] parsed = CSVParser.parse(s);
43      assertTrue(parsed.length == 3);
44      assertTrue(parsed[2].length == 4);
45      assertTrue(parsed[1][2].equals("the"));
46      assertTrue(parsed[2][1].equals("blah"));
47    }
48  
49    /**
50     * Tests strings with no data in certain cells.
51     */
52    @Test
53    public void consecutiveCommasTest() {
54      String s = ",there are so,\n"
55          + "many,,commas\n"
56          + "in this string,,";
57      String[][] parsed = CSVParser.parse(s);
58      assertTrue(parsed.length == 3);
59      assertTrue(parsed[1].length == 3);
60      assertTrue(parsed[0][0].equals(""));
61      assertTrue(parsed[2][2].equals(""));
62      assertTrue(parsed[2][0].equals("in this string"));
63    }
64  
65    /**
66     * Tests numbers and decimals are supported.
67     */
68    @Test
69    public void numbersAndDecimalsTest() {
70      String s = "3.1415,yes,...\n"
71          + "4,10203f,3";
72      String[][] parsed = CSVParser.parse(s);
73      assertTrue(parsed.length == 2);
74      assertTrue(parsed[0][2].equals("..."));
75    }
76  
77    /**
78     * Tests that jagged files are correctly caught.
79     */
80    @Test(expected = AssertionError.class)
81    public void jaggedCSVTest() {
82      String s = "one,two,three\n"
83          + "one,two,three\n"
84          + "one,two\n";
85      CSVParser.parse(s);
86    }
87  }
